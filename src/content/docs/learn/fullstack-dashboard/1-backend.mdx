## Imports and Client Setup

```js title="backend/index.js"
import { createWalletClient, http } from '@arkiv-network/sdk';
import { kaolin } from '@arkiv-network/sdk/chains';
import { privateKeyToAccount } from '@arkiv-network/sdk/accounts';
import { ExpirationTime, jsonToPayload } from '@arkiv-network/sdk/utils';
import axios from 'axios';

const privateKey = process.env.PRIVATE_KEY;
if (!privateKey) {
  throw new Error('PRIVATE_KEY is not set in the .env file.');
}

const account = privateKeyToAccount(privateKey);

const client = createWalletClient({
  chain: kaolin,
  transport: http(),
  account,
});

console.log("Backend service connected as: " + client.account.address);
```

This creates a `walletClient` authenticated with a private key — the standard pattern for server-side Arkiv usage.

## Fetching Data from CoinGecko

```js title="backend/index.js"
const params = new URLSearchParams({
  vs_currency: 'usd',
  ids: 'bitcoin,ethereum,golem',
  sparkline: 'false',
});

const COINGECKO_URL = "https://api.coingecko.com/api/v3/coins/markets?" + params;

async function fetchCryptoData() {
  try {
    const response = await axios.get(COINGECKO_URL);
    console.log('Successfully fetched data from CoinGecko.');
    return response.data;
  } catch (error) {
    console.error('Error fetching data from CoinGecko:', error.message);
    return [];
  }
}
```

## Uploading Data to Arkiv

This is the core of the backend. It takes crypto data and creates entities in a **single batch transaction** using `mutateEntities()`:

```js title="backend/index.js"
async function uploadDataToArkiv(cryptoData) {
  if (cryptoData.length === 0) {
    console.log("No crypto data to upload.");
    return;
  }

  try {
    const createPayloads = cryptoData.map(tokenData => {
      const {
        id,
        current_price,
        market_cap,
        price_change_percentage_24h
      } = tokenData;
      return {
        payload: jsonToPayload({
          price: current_price,
          marketCap: market_cap,
          change24h: price_change_percentage_24h,
          timestamp: Date.now(),
        }),
        contentType: 'application/json',
        attributes: [
          { key: 'token', value: id },
        ],
        expiresIn: ExpirationTime.fromHours(3),
      };
    });

    const result = await client.mutateEntities({
      creates: createPayloads
    });

    result.createdEntities.forEach((entityKey, index) => {
      const tokenId = cryptoData[index].id;
      console.log(`Created entity for ${tokenId}. Key: ${entityKey}`);
    });
  } catch (error) {
    console.error('Failed to create entities:', error.message);
  }
}
```

<Aside>
Each token snapshot expires after 3 hours. The dashboard queries the most recent snapshots — older ones are automatically cleaned up by Arkiv.
</Aside>

**Key patterns demonstrated:**
- `mutateEntities({ creates: [...] })` for batch creation instead of individual `createEntity` calls
- `jsonToPayload()` to encode structured data
- `ExpirationTime.fromHours(3)` for time-scoped storage
- `token` attribute for querying by cryptocurrency type

## The Main Loop

```js title="backend/index.js"
async function runUpdateCycle() {
  console.log('\n--- Starting new update cycle ---');
  const cryptoData = await fetchCryptoData();

  if (cryptoData.length > 0) {
    await uploadDataToArkiv(cryptoData);
  }
}

runUpdateCycle();
setInterval(runUpdateCycle, 60000);
```
