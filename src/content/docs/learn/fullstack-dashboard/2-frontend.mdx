## Frontend Query Script

This is where the frontend queries Arkiv directly from the browser — no server needed for reads. The SDK is loaded via CDN (ESM imports):

```js title="frontend/script.js"
const USER_ADDRESS = '0xYourAddressHere';

import { createPublicClient, http } from 'https://esm.sh/@arkiv-network/sdk@0.6.0?target=es2022&bundle-deps';
import { eq } from 'https://esm.sh/@arkiv-network/sdk@0.6.0/query?target=es2022&bundle-deps';
import { kaolin } from 'https://esm.sh/@arkiv-network/sdk@0.6.0/chains?target=es2022&bundle-deps';
import { initializeCharts, updateChart, updateMarketCapChart, updatePriceBoxes, getCharts } from './charts.js';

const client = createPublicClient({
  chain: kaolin,
  transport: http(),
});
```

<Aside type="caution">
Replace `0xYourAddressHere` with the wallet address from your backend's private key. This filters entities to only those created by your backend.
</Aside>

### Querying Token Data

Each token is queried separately by its `token` attribute, filtered to entities owned by the backend wallet:

```js title="frontend/script.js"
async function fetchTokenDataFromArkiv(tokenId) {
  const query = client.buildQuery();
  const result = await query
    .where(eq('token', tokenId))
    .ownedBy(USER_ADDRESS)
    .withPayload(true)
    .fetch();

  return result.entities
    .map((entity) => ({ ...entity.toJson(), entityKey: entity.key }))
    .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());
}
```

**Key patterns:**
- `createPublicClient` — read-only, no private key needed in the browser
- `.where(eq('token', tokenId))` — filter by the attribute set in the backend
- `.ownedBy(USER_ADDRESS)` — only show entities from our backend
- `.withPayload(true)` — include the JSON data (price, marketCap, etc.)

### Polling for Updates

The UI refreshes every 15 seconds by querying all three tokens in parallel:

```js title="frontend/script.js"
async function updateUI() {
  try {
    const [btcData, ethData, glmData] = await Promise.all([
      fetchTokenDataFromArkiv('bitcoin'),
      fetchTokenDataFromArkiv('ethereum'),
      fetchTokenDataFromArkiv('golem'),
    ]);

    if (btcData.length > 0 || ethData.length > 0 || glmData.length > 0) {
      updatePriceBoxes(btcData, ethData, glmData);
      updateMarketCapChart(btcData, ethData, glmData);

      const { btcPriceChart, ethPriceChart, glmPriceChart } = getCharts();
      if (btcData.length > 0) updateChart(btcPriceChart, btcData, 'price');
      if (ethData.length > 0) updateChart(ethPriceChart, ethData, 'price');
      if (glmData.length > 0) updateChart(glmPriceChart, glmData, 'price');
    }
  } catch (error) {
    console.error('Error updating UI:', error);
  }
}

window.addEventListener('DOMContentLoaded', () => {
  initializeCharts();
  updateUI();
  setInterval(updateUI, 15000);
});
```
